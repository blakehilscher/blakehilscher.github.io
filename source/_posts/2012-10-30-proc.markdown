```ruby
class ObjectContext
   
  def initialize
    @number = 10
  end
   
  def evaluate(proc)
    instance_exec &proc
  end
   
end
 
# our multiplication proc
t = -> { puts( @number * @number) }
 
# evaluate in current context
@number = 5
t.call #=> 25 
 
# evaluate in context of an ObjectFactory instance 
o = ObjectContext.new
o.evaluate(t) #=> 100
```
